const canvas = document.getElementById("matrix"); const ctx = canvas.getContext("2d"); canvas.width = window.innerWidth; canvas.height = window.innerHeight; const particles = []; class Particle { constructor() { this.x = Math.random() * canvas.width; this.y = Math.random() * canvas.height; this.size = Math.random() * 2 + 1; this.speedY = Math.random() * 0.6 + 0.3; this.alpha = Math.random() * 0.5 + 0.3; } update() { this.y += this.speedY; if (this.y > canvas.height) { this.y = 0; this.x = Math.random() * canvas.width; } } draw() { ctx.fillStyle = `rgba(255, 20, 147, ${this.alpha})`; ctx.shadowBlur = 10; ctx.shadowColor = "#ff00dd"; ctx.fillRect(this.x, this.y, this.size, this.size * 5); } } function initParticles() { for (let i = 0; i < 200; i++) { particles.push(new Particle()); } } function animate() { ctx.clearRect(0, 0, canvas.width, canvas.height); particles.forEach((p) => { p.update(); p.draw(); }); requestAnimationFrame(animate); } initParticles(); animate(); window.onresize = () => { canvas.width = window.innerWidth; canvas.height = window.innerHeight; };
